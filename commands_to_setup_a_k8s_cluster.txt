3 × Ubuntu 22.04 LTS machines (bare metal or VMs)
1 control-plane node — runs the API server, scheduler, controller-manager, and etcd
2 worker nodes — run application pods
CNI: Flannel using its default Pod network 10.244.0.0/16

Disable Swap (Run on all 3 machines)
====================================
sudo swapoff -a
sudo sed -i '/ swap / s/^/#/' /etc/fstab
Verification:
	free -h (swap should be 0B)

Load the required kernel modules (Run on all 3 machines)
=======================================================

# Load required kernel modules for container networking
sudo modprobe overlay
sudo modprobe br_netfilter
 
# Persist them across reboots
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

Verification:
	lsmod | grep overlay
	lsmod | grep br_netfilter

Networking sysctl (Run on all 3 machines)
=========================================

# Allow iptables to inspect bridged & IPv6 traffic; enable IP forwarding
cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF
sudo sysctl --system   # apply without rebooting

Verification:
	cat /etc/sysctl.d/k8s.conf

Install containerd (Run on all 3 machines)
===========================================

# Install containerd — the CRI runtime that actually runs pods
sudo apt-get update && sudo apt-get install -y containerd
 
# Generate default config and switch to systemd cgroups
sudo mkdir -p /etc/containerd
containerd config default | sudo tee /etc/containerd/config.toml

sudo sed -i 's/SystemdCgroup = false/SystemdCgroup = true/' /etc/containerd/config.toml
sudo systemctl restart containerd && sudo systemctl enable containerd


Verify containerd is installed - containerd --version
Verify service is running - systemctl status containerd, systemctl is-enabled containerd
Verify it starts on boot - cat /etc/containerd/config.toml | grep SystemdCgroup

Install kubeadm, kubelet, and kubectl(Run on all 3 machines)
============================================================
 
# Tools needed to talk to the Kubernetes apt repo over HTTPS
sudo apt-get install -y apt-transport-https ca-certificates curl gpg
 
# Add the Kubernetes signing key (v1.30 stream — pick the version you want)
curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.30/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg

# Register the Kubernetes apt repository
sudo mkdir -p /etc/apt/keyrings

echo 'deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.30/deb/ /' | sudo tee /etc/apt/sources.list.d/kubernetes.list

# Install the three CLIs
sudo apt-get update
sudo apt-get install -y kubelet kubeadm kubectl
 
# Pin them so apt-get upgrade can't bump versions silently
sudo apt-mark hold kubelet kubeadm kubectl

Verification:

	kubeadm version
	kubelet --version
	kubectl version

Initialize the control plane (Only on control plane)@@@
====================================================

# Initialise the control plane; pick a Pod CIDR that matches your CNI
sudo kubeadm init --pod-network-cidr=10.244.0.0/16

#Set up kubectl access

mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

Verification:
	kubectl cluster-info
	kubectl get nodes -o wide
	kubectl describe node <your-node-name> | grep -i podcidr

Install Flannel as CNI (Only on control plane) @@@@
==============================================

# Install Flannel — the CNI that wires up pod networking
kubectl apply -f https://github.com/flannel-io/flannel/releases/latest/download/kube-flannel.yml

Verification:
	kubectl get pods -n kube-flannel
	kubectl get ds -n kube-flannel

Join the worker nodes (Only on the worker nodes) 
=================================================

# Use the join command printed by 'kubeadm init' on the control plane
From the control plane, run 
	kubeadm token create --print-join-command

sudo kubeadm join <control-plane-ip>:6443 --token <token> --discovery-token-ca-cert-hash sha256:<hash>
